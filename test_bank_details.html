Test Script for Console

Copy và paste các lệnh sau vào Console (F12):

// 1. Kiểm tra element có tồn tại không
const bankDetails = document.getElementById('bankDetails');
console.log('bankDetails element:', bankDetails);

// 2. Kiểm tra computed style
if (bankDetails) {
    const style = window.getComputedStyle(bankDetails);
    console.log('Display:', style.display);
    console.log('Visibility:', style.visibility);
    console.log('Opacity:', style.opacity);
    console.log('Height:', style.height);
}

// 3. Kiểm tra accountNumber input
const accountInput = document.getElementById('accountNumber');
console.log('accountNumber element:', accountInput);
if (accountInput) {
    const inputStyle = window.getComputedStyle(accountInput);
    console.log('Input Display:', inputStyle.display);
    console.log('Input Visibility:', inputStyle.visibility);
}

// 4. Force hiển thị
if (bankDetails) {
    bankDetails.classList.remove('hidden');
    bankDetails.style.cssText = 'display: block !important; visibility: visible !important; opacity: 1 !important; height: auto !important;';
    console.log('Đã force hiển thị bankDetails');
}

if (accountInput) {
    accountInput.style.cssText = 'display: block !important; visibility: visible !important; width: 100% !important;';
    console.log('Đã force hiển thị accountInput');
}

// 5. Kiểm tra parent elements
if (bankDetails) {
    let parent = bankDetails.parentElement;
    let level = 0;
    while (parent && level < 5) {
        const parentStyle = window.getComputedStyle(parent);
        console.log(`Parent level ${level}:`, parent.tagName, parent.className);
        console.log(`  - Display: ${parentStyle.display}`);
        console.log(`  - Visibility: ${parentStyle.visibility}`);
        parent = parent.parentElement;
        level++;
    }
}
